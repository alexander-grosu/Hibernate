import java.util.ArrayList;

public class exemple {

    public static void main(String[] args) {
        //Creating a generic ArrayList
        ArrayList<String> arlTest = new ArrayList<String>();

        System.out.println("ArrayList size: " + arlTest.size());
        //Output:  0
        System.out.println("ArrayList: " + arlTest);
        //Output:  []
        arlTest.add("D");
        arlTest.add("U");
        arlTest.add(0, "k");
        arlTest.remove(2);
        System.out.println("ArrayList size: " + arlTest.size() + "\nArrayList: " + arlTest);
        //Output: 2
        //Output: [k, D]
        arlTest.add("xyzw");
        arlTest.add(1, "ssss");
        System.out.println(arlTest);
        //Output: [k, ssss, D, xyzw]   ("ssss" занял индекс 1 а "D" поменял индекс с 1 на 2)
        for (String item : arlTest) {
            System.out.println(item);
        }
        //Output: k
        //        ssss
        //        D
        //        xyzw
        arlTest.sort(String::compareTo);
        System.out.println(arlTest);
        //Output: [D, k, ssss, xyzw]    sort() - выводит arrayList в алфавитном порядке, сортирует.
    }
}
